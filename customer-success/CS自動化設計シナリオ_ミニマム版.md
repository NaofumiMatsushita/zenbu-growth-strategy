```javascript
// NPS自動送信スクリプト

function sendNPSSurvey() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet()
    .getSheetByName('顧客リスト');
  const data = sheet.getDataRange().getValues();

  // ヘッダー行をスキップ
  for (let i = 1; i < data.length; i++) {
    const row = data[i];
    const email = row[1];           // B列: メールアドレス
    const name = row[0];            // A列: 顧客名
    const completionDate = row[5];  // F列: 測定完了日
    const surveySent = row[6];      // G列: アンケート送信済み

    // 測定完了から7日後 & 未送信
    const today = new Date();
    const daysSince = (today - completionDate) / (1000 * 60 * 60 * 24);

    if (daysSince >= 7 && daysSince < 8 && !surveySent) {
      // メール送信
      const subject = `${name}様、ZENBUのご利用はいかがでしたか？`;
      const body = `
${name}様

先日はZENBUをご利用いただき、ありがとうございました。

簡単なアンケートにご協力いただけますと幸いです。
所要時間は1分程度です。

アンケートはこちら:
https://forms.gle/xxxxx

今後ともZENBUをよろしくお願いいたします。

ZENBU株式会社
カスタマーサクセスチーム
      `;

      MailApp.sendEmail(email, subject, body);

      // 送信済みフラグを立てる
      sheet.getRange(i + 1, 7).setValue('送信済み');

      Logger.log(`NPS送信: ${email}`);
    }
  }
}

// 毎日午前9時に自動実行（トリガー設定）
```

    'channel': channel,

    'method': 'post',
    'contentType': 'application/json',

  };

// NPS批判者を検知したら即通知
function checkNPSResponses() {
